class SystemConfigValidator:
    """Validates system configuration against hardware specs and installation plan"""
    def __init__(self):
        self.hardware_profile = {
            "cpu": "AMD Ryzen 7 5700U",
            "gpu": "AMD Radeon Lucienne",
            "ram": "32GB",
            "storage": {
                "main": "256GB NVMe",
                "partitions": {
                    "efi": "/dev/nvme0n1p1",
                    "system": "/dev/nvme0n1p3",
                    "legacy_linux": "/dev/nvme0n1p4"
                }
            }
        }
        
        self.installation_blueprint = {
            "target_layout": {
                "system_partition": {"size": "2GB", "mount": "/boot"},
                "root_partition": {"size": "118GB", "mount": "/"},
                "projects_partition": {"size": "10GB", "mount": "/mnt/projects"},
                "home_partition": {"size": "70GB", "mount": "/home"},
                "library_partition": {"size": "46GB", "mount": "/library"}
            },
            "llm_hierarchy": ["distilBERT", "KriKri-7B"]
        }

    def validate_hardware(self):
        """Check against documented hardware specs"""
        return {
            "compatibility": {
                "ubuntu_24.04": self._check_amd_compatibility(),
                "gpu_acceleration": self._check_radeon_drivers(),
                "storage_plan": self._verify_partition_sizes()
            }
        }

    def _check_amd_compatibility(self):
        """Verify AMD processor support"""
        return "Zen 2" in self.hardware_profile["cpu"]

    def _check_radeon_drivers(self):
        """Check for required GPU drivers"""
        try:
            output = subprocess.check_output("lspci -k | grep -A3 VGA", shell=True)
            return "amdgpu" in output.decode()
        except subprocess.CalledProcessError:
            return False

    def _verify_partition_sizes(self):
        """Validate against installation plan"""
        required = self.installation_blueprint["target_layout"]
        return {
            "system": required["system_partition"]["size"],
            "root": required["root_partition"]["size"],
            "projects": required["projects_partition"]["size"]
        }

class InstallationManager:
    """Manages Ubuntu 24.04 installation workflow"""
    def __init__(self):
        self.steps = [
            "Backup verification",
            "ISO preparation",
            "GRUB configuration",
            "Partition resizing",
            "Base system installation",
            "Data migration",
            "Final partitioning"
        ]
        
        self.grub_template = """menuentry "Ubuntu 24.04 ISO" {{
    set isofile="/ubuntu-24.04-desktop-amd64.iso"
    loopback loop (hd0,5)$isofile
    linux (loop)/casper/vmlinuz boot=casper iso-scan/filename=$isofile quiet splash --
    initrd (loop)/casper/initrd

    def generate_install_script(self):
        """Create automated installation steps"""
        return {
            "prepare_iso": "sudo cp ~/Downloads/ubuntu-24.04-desktop-amd64.iso /mnt/bootdrive/",
            "configure_grub": f"echo '{self.grub_template}' | sudo tee /etc/grub.d/40_custom",
            "update_grub": "sudo update-grub",

        """Generate partition resizing instructions"""

            "resize /dev/nvme0n1p3 2GB",
            "create /dev/nvme0n1p6 ext4 label='System'",
            "format /dev/nvme0n1p6 ext4",
            "mount /dev/nvme0n1p6 /boot"
        ]

class SystemContextValidator:
    """Advanced hardware/context validator (New)"""
    def __init__(self):
        self.expected_specs = {
            'cpu': 'AMD Ryzen 7 5700U',
            'gpu': 'AMD Radeon Lucienne',
            'ram': '32GB DDR4',
            'storage': '256GB NVMe'
        }
        
    def verify_context(self, conversation_context):
        """Compare conversation context with expected specs"""
        discrepancies = []
        for component, expected in self.expected_specs.items():
            mentioned = conversation_context.get(component, '').lower()
            if mentioned and expected.lower() not in mentioned:
                discrepancies.append(f"{component.upper()} mismatch: {expected} vs {mentioned}")
        return discrepancies

    def get_hardware_actual(self):
        """Retrieve actual hardware specs"""
        return {
            'cpu': self._get_cpu_info(),
            'gpu': self._get_gpu_info(),
            'ram': self._get_ram_info(),
            'storage': self._get_storage_info()
        }

    def _get_cpu_info(self):
        try:
            return subprocess.getoutput("lscpu | grep 'Model name'").split(':')[1].strip()
        except:
            return "AMD Ryzen 7 5700U"

    def _get_gpu_info(self):
        try:
            return subprocess.getoutput("lspci | grep VGA").split(':')[2].strip()
        except:
            return "AMD Radeon Lucienne"

    def _get_ram_info(self):
        try:
            return subprocess.getoutput("grep MemTotal /proc/meminfo").split()[1] + " KB"
        except:
            return "32GB DDR4"

    def _get_storage_info(self):
        try:
            return subprocess.getoutput("lsblk -o SIZE,MODEL | grep NVMe").strip()
        except:
            return "256GB NVMe"

class HardwareAwareOrchestrator:
    """Context-aware system orchestrator (New)"""
    def __init__(self):
        self.validator = SystemContextValidator()
        self.context_history = []
    
    def process_query(self, query, conversation_context):
        """Handle query with hardware context"""
        actual_hardware = self.validator.get_hardware_actual()
        context_issues = self.validator.verify_context(conversation_context)
        
        self.context_history.append({
            'timestamp': datetime.now().isoformat(),
            'query': query,
            'context': conversation_context,
            'validation_issues': context_issues
        })
        
        return {
            'recommendations': self._generate_recommendations(query, actual_hardware),
            'context_warnings': context_issues
        }

    def _generate_recommendations(self, query, hardware):
        """Create hardware-specific recommendations"""
        recs = []
        if 'ryzen' in hardware['cpu'].lower():
            recs.append("Enable AMD P-State driver for better power management")
        if 'radeon' in hardware['gpu'].lower():
            recs.append("Configure amdgpu driver for Vulkan support")
        if 'storage' in query.lower():
            recs.append("Consider NVMe optimization for vector database workloads")
        return recs
