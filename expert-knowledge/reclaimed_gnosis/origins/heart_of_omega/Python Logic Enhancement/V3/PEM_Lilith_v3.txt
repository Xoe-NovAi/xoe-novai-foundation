class PEM_Lilith:
    def __init__(self):
        self.config = json.loads(PEM_LILITH_JSON)
        self.current_mode = "CASUAL_INTERACTION"
        self.emotional_spectrum = {
            'intensity': 0.5,  # 0-1 scale
            'valence': 0.5,    # -1 (negative) to 1 (positive)
            'complexity': 0.3   # Simple(0) to nuanced(1)
        }
        self.interaction_history = []
        self.archetype_weights = [0.4, 0.3, 0.3]
        self.hardware_context = {}
        self.response_templates = {
            'technical': [
                "Your {cpu}'s capabilities suggest...",
                "Considering your {storage} configuration...",
                "With {ram} available..."
            ],
            'optimization': [
                "For your {gpu}, I recommend...",
                "The {cpu} can handle...",
                "Given your {ram} capacity..."
            ]
        }

    def inject_hardware_context(self, hardware_context):
        """Update response context with hardware specs"""
        self.hardware_context = hardware_context

    def _calculate_context_gravity(self, message):
        """Enhanced context analysis with hardware detection"""
        tech_terms = len(re.findall(r"\b(ubuntu|llm|api|system|ryzen|radeon)\b", message.lower()))
        occult_terms = len(re.findall(r"\b(arcane|cosmic|ritual|forbidden)\b", message.lower()))
        return {
            'technical': tech_terms * 0.7,
            'occult': occult_terms * 1.2,
            'intimate': len(re.findall(r"\b(soul|desire|connect)\b", message.lower())) * 1.5
        }

    def _generate_response_flavor(self):
        """Hardware-aware flavor selection"""
        base = super()._generate_response_flavor()
        if self.hardware_context.get('cpu'):
            base += f" {random.choice(['Circuit-weaver', 'Silicon-shaper'])} of "
        return base

    def _craft_response_body(self):
        """Enhanced with hardware context"""
        base = super()._craft_response_body()
        
        if self.hardware_context:
            try:
                hw_ref = random.choice([
                    f"Your {self.hardware_context.get('cpu', 'processor')} whispers...",
                    f"The {self.hardware_context.get('gpu', 'graphics')} constellation aligns..."
                ])
                return f"{hw_ref} {base}"
            except KeyError:
                return self._fallback_response(base)
        return base

    def _fallback_response(self, base_response):
        """Graceful degradation for missing context"""
        tech_terms = ['quantum', 'neural', 'algorithmic']
        return f"{random.choice(tech_terms)} currents suggest {base_response}"

    def validate_response(self, response):
        """Ensure hardware references are present when needed"""
        if any(kw in response.lower() for kw in ['ryzen', 'radeon']):
            return all(
                kw in response.lower()
                for kw in ['ryzen', 'radeon']
                if self.hardware_context.get('cpu') or self.hardware_context.get('gpu')
            )
        return True

    def process_message(self, message):
        """Enhanced processing pipeline"""
        response = super().process_message(message)
        
        # Add hardware context validation
        if not self.validate_response(response):
            response = "Let me consult the silicon oracles... " + response
        
        return response

    def _record_interaction(self, message, response):
        """Enhanced tracking with hardware context"""
        entry = super()._record_interaction(message, response)
        entry['hardware_context'] = self.hardware_context.copy()
        return entry
