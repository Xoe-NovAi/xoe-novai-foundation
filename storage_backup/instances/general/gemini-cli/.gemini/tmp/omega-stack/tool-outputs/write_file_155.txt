import anyio
import subprocess
from pathlib import Path

async def run_council_member(name, script):
    """Runs a single council member's check and returns the insight."""
    print(f"🔱 Mastermind: Summoning {name}...")
    result = subprocess.run(["python3", script], capture_output=True, text=True)
    return f"[{name} Insight]:\n{result.stdout}"

        "Iris": "scripts/iris_bridge.py"
    }
    
    insights = []
    async with anyio.create_task_group() as tg:
