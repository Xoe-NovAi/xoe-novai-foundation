FROM xnai-base:latest

LABEL maintainer="Xoe-NovAi Team"
LABEL version="0.1.8"
LABEL description="Xoe-NovAi Memory Bank MCP Service"

# Ensure foundation modules are in the search path
ENV PYTHONPATH=/app

# Set working directory to the MCP server location
WORKDIR /app/mcp-server

# Install service-specific dependencies not in base
RUN uv pip install --system --no-cache \
    mcp>=0.1.0 \
    sqlalchemy>=2.0 \
    aiosqlite>=3.13.0 \
    aiohttp>=3.13.3

# Copy application code (app is already in base context, but ensuring mcp-server is here)
COPY mcp-servers/memory-bank-mcp/ /app/mcp-server/

# Health check
HEALTHCHECK --interval=30s --timeout=10s --retries=3 \
    CMD python -c "import sys; sys.exit(0)" || exit 1

# Run the server
CMD ["python", "server.py"]
